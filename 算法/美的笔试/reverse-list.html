ListNode dummy = new ListNode(0); dummy.next = head;
ListNode pre = dummy;
for (int i = 1; i < m; i++) pre = pre.next; // pre 移到第 m-1 个节点
ListNode cur = pre.next; // cur 固定在第 m 个节点
for (int i = 0; i < n - m; i++) {
    ListNode next = cur.next; // ① 取出 cur 后面的节点
    cur.next = next.next; // ② cur 跳过 next
    next.next = pre.next; // ③ next 指向旧的第一个
    pre.next = next;     // ④ pre 指向 next（头插完成）
}
return dummy.next;
