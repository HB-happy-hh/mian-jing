主函数 — 遍历整张地图
public int numIslands(char[][] grid) {
  int count = 0;
  for (int i = 0; i < grid.length; i++) {
    for (int j = 0; j < grid[0].length; j++) {
      if (grid[i][j] == '1') {
        count++;        // 发现新岛屿
        dfs(grid, i, j); // 淹掉整座岛
      }
    }
  }
  return count;
}
DFS 递归函数 — 淹没相连陆地
void dfs(char[][] grid, int i, int j) {
  // 越界 或 是水 → 停止
  if (i < 0 || i >= grid.length
   || j < 0 || j >= grid[0].length
   || grid[i][j] == '0') {
    return;
  }
  grid[i][j] = '0'; // 标记已访问（淹掉）
  dfs(grid, i+1, j); // 往下
  dfs(grid, i-1, j); // 往上
  dfs(grid, i, j+1); // 往右
  dfs(grid, i, j-1); // 往左
}
