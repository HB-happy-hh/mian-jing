class Solution {
            List<String> result = new ArrayList<>();
    public List<String> restoreIpAddresses(String s) {

        dfs(s,0,new ArrayList<>());
        return result;
    }

    public void dfs(String s, int index ,List<String> segment){
        if(segment.size()==4 && index==s.length()){
            result.add(String.join(".",segment));
            return;
        }

        if(segment.size()==4 || index==s.length()) return;

        for(int len =1; len<=3;len++){
            if((index + len)>s.length()) break;

            String seg = s.substring(index, index+len);

            if(seg.length()>1 && seg.charAt(0)=='0') break;
            if(Integer.parseInt(seg)>255) break;

            segment.add(seg);
            dfs(s,index+len,segment);
            segment.remove(segment.size()-1);

        }
    }
}
