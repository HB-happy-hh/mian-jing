public int findKthLargest(int[] nums, int k) {
    // 小顶堆：堆顶永远是堆内最小的元素
    PriorityQueue<Integer> minHeap = new PriorityQueue<>();

    for (int num : nums) {
        minHeap.offer(num);          // 先入堆

        if (minHeap.size() > k) {
            minHeap.poll();           // 堆满，弹出最小值（不可能是第k大）
        }
    }
    // 堆中剩下的 k 个，就是最大的 k 个
    // 堆顶是这 k 个里最小的，即第 k 大
    return minHeap.peek();
}
